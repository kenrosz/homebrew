138#if defined(_MSC_VER)
139#include <intrin.h>
140static INLINE long opj_lrintf(float f)
141{
142#ifdef _M_X64
143 return _mm_cvt_ss2si(_mm_load_ss(&f));
144
145 /* commented out line breaks many tests */
146 /* return (long)((f>0.0f) ? (f + 0.5f):(f -0.5f)); */
147#elif defined(_M_IX86)
148 int i;
149 _asm{
150 fld f
151 fistp i
152 };
153
154 return i;
155#else
156 return (long)((f>0.0f) ? (f + 0.5f) : (f - 0.5f));
157#endif
158}
159#elif defined(__BORLANDC__)
160static INLINE long opj_lrintf(float f)
161{
162#ifdef _M_X64
163 return (long)((f > 0.0f) ? (f + 0.5f) : (f - 0.5f));
164#else
165 int i;
166
167 _asm {
168 fld f
169 fistp i
170 };
171
172 return i;
173#endif
174}
175#else
176static INLINE long opj_lrintf(float f)

178 return lrintf(f);
